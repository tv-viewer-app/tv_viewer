### Issue #28: Israel Country Classification (3 days)

- [ ] Create country code mapper
  ```dart
  // lib/utils/country_code_mapper.dart
  class CountryCodeMapper {
    static const Map<String, String> isoToName = {
      'IL': 'Israel',
      'US': 'United States',
      'GB': 'United Kingdom',
      'CA': 'Canada',
      'AU': 'Australia',
      'DE': 'Germany',
      'FR': 'France',
      'IT': 'Italy',
      'ES': 'Spain',
      'NL': 'Netherlands',
      // Add top 50 countries
    };
    
    static String normalize(String? country) {
      if (country == null || country.isEmpty) return 'Unknown';
      
      // Try ISO code lookup
      final normalized = isoToName[country.toUpperCase()];
      if (normalized != null) return normalized;
      
      // Try reverse lookup (name → name)
      final lowerCountry = country.toLowerCase();
      for (final entry in isoToName.entries) {
        if (entry.value.toLowerCase() == lowerCountry) {
          return entry.value;
        }
      }
      
      // Capitalize first letter of each word
      return country.split(' ')
          .map((word) => word.isEmpty ? '' : 
                word[0].toUpperCase() + word.substring(1).toLowerCase())
          .join(' ');
    }
  }
  ```

  ```dart
  // lib/services/m3u_service.dart
  static String normalizeUrl(String url) {
    try {
      final uri = Uri.parse(url);
      
      // Normalize scheme (prefer https)
      final scheme = uri.scheme == 'http' ? 'https' : uri.scheme;
      
      // Remove trailing slash
      var path = uri.path;
      if (path.endsWith('/') && path.length > 1) {
        path = path.substring(0, path.length - 1);
      }
      
      // Remove common session parameters
      final cleanQuery = Map<String, dynamic>.from(uri.queryParameters)
        ..removeWhere((key, value) => 
            key.toLowerCase().contains('token') ||
            key.toLowerCase().contains('session') ||
            key.toLowerCase().contains('timestamp'));
      
      return Uri(
        scheme: scheme,
        host: uri.host.toLowerCase(), // Normalize host case
        port: uri.port == 80 || uri.port == 443 ? null : uri.port,
        path: path,
        queryParameters: cleanQuery.isEmpty ? null : cleanQuery,
      ).toString();
    } catch (e) {
      logger.warning('URL normalization failed for: $url', e);
      return url;
    }
  }
  ```

- [ ] Update de-duplication logic
  ```dart
  // lib/services/m3u_service.dart (in fetchAllChannels)
  final allChannels = <Channel>[];
  final seenUrls = <String>{};
  int duplicateCount = 0;
  
  for (final channel in channels) {
    final normalizedUrl = normalizeUrl(channel.url);
    
    if (!seenUrls.contains(normalizedUrl)) {
      seenUrls.add(normalizedUrl);
      allChannels.add(channel);
    } else {
      duplicateCount++;
      logger.debug('Duplicate removed: ${channel.name} - $normalizedUrl');
    }
  }
  
  logger.info('De-duplication complete: '
    '${allChannels.length} unique channels, '
    '$duplicateCount duplicates removed');
  ```

- [ ] Write unit tests
